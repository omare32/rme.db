SELECT INITCAP(PARAMETER_TITLE) PARAMETER_TITLE, PARAMETER_VALUE 
FROM 
( 
 SELECT 'Supplier' PARAMETER_TITLE,to_char((SELECT VENDOR_NAME FROM AP_SUPPLIERS WHERE VENDOR_ID = :P_VENDOR_ID)) PARAMETER_VALUE ,3 SERIAL FROM DUAL 
 UNION
 SELECT 'CURRENCY CODE' PARAMETER_TITLE,to_char(:P_CURRENCY_CODE) PARAMETER_VALUE ,15 SERIAL FROM DUAL 
 UNION
SELECT 'Department' PARAMETER_TITLE,to_char(:p_dep) PARAMETER_VALUE ,1 SERIAL FROM DUAL 
 UNION
 SELECT 'From GL Date' PARAMETER_TITLE,to_char(:P_FROM_GL_DATE) PARAMETER_VALUE ,20 SERIAL FROM DUAL 
UNION
 SELECT 'To GL Date' PARAMETER_TITLE,to_char(:P_TO_GL_DATE) PARAMETER_VALUE ,20 SERIAL FROM DUAL 
UNION
 SELECT 'Include Zero Net Amount' PARAMETER_TITLE,to_char(decode(:P_INCLUDE_ZERO_NET,'Y', 'Yes', 'No')) PARAMETER_VALUE ,25 SERIAL FROM DUAL
UNION
 SELECT 'Summary' PARAMETER_TITLE,to_char(decode(:P_SUMMARY,'Y', 'Yes', 'No')) PARAMETER_VALUE ,25 SERIAL FROM DUAL 
UNION
 SELECT 'Group By Invoice Type' PARAMETER_TITLE,to_char(decode(:P_GROUP_INVOICE_TYPE,'Y', 'Yes', 'No')) PARAMETER_VALUE ,25 SERIAL FROM DUAL
UNION
 SELECT 'Invoice Type' PARAMETER_TITLE,to_char(:P_INV_TYPE) PARAMETER_VALUE ,30 SERIAL FROM DUAL
UNION
 SELECT 'Project Name' PARAMETER_TITLE,to_char((SELECT NAME FROM PA_PROJECTS_ALL WHERE PROJECT_ID = :P_PROJECT_ID)) PARAMETER_VALUE ,35 SERIAL FROM DUAL

--P_GROUP_VENDOR_TYPE
 )
ORDER BY SERIAL
