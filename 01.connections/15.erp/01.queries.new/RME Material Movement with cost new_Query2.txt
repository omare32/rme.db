SELECT INITCAP(PARAMETER_TITLE) PARAMETER_TITLE, PARAMETER_VALUE 
FROM 
( 

 SELECT 'FROM DATE' PARAMETER_TITLE,to_char(:P_FROM_DATE,'DD.MM.YYYY') PARAMETER_VALUE ,1 SERIAL FROM DUAL 
 UNION
 SELECT 'TO DATE' PARAMETER_TITLE,to_char(:P_TO_DATE,'DD.MM.YYYY')  PARAMETER_VALUE ,4 SERIAL FROM DUAL 


 )
ORDER BY SERIAL
