SELECT INITCAP(PARAMETER_TITLE) PARAMETER_TITLE, PARAMETER_VALUE 
FROM 
( 
 SELECT 'Projects' PARAMETER_TITLE,:P_PROJECT  PARAMETER_VALUE ,1 SERIAL FROM DUAL 
 UNION
 SELECT 'Book' PARAMETER_TITLE,:P_BOOK PARAMETER_VALUE ,5 SERIAL FROM DUAL 
 UNION
 SELECT 'From Period' PARAMETER_TITLE,:p_period PARAMETER_VALUE ,10 SERIAL FROM DUAL 
 UNION
 SELECT 'With Cost Center' PARAMETER_TITLE,:dep PARAMETER_VALUE ,15 SERIAL FROM DUAL
 UNION
 SELECT 'Date Placed from' PARAMETER_TITLE,TO_CHAR(:p_plced_from) PARAMETER_VALUE ,20 SERIAL FROM DUAL
 UNION
 SELECT 'Date Placed To' PARAMETER_TITLE,TO_CHAR(:p_plced_to) PARAMETER_VALUE ,20 SERIAL FROM DUAL  
)
ORDER BY SERIAL
